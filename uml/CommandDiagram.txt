@startuml
package command {
    class AddItemCommand {
        # commandWord = "addItem"
        - price: Double
        - quantity: Int
        - itemName: String
    }

    class AddQuoteCommand {
        # commandWord = "add"
        - quoteName: String
        - customerName: String
    }

    class DeleteItemCommand {
        # commandWord = "deleteItem"
        - itemName: String
    }

    class ExitCommand {
        # commandWord = "exit"
        + exit(): true
    }

    class RegisterCommand {
        # commandWord = "register"
        - newName: String
    }

    class CalculateTotalCommand {
        # commandWord = "total"
    }

    class FinishQuoteCommand {
        # commandWord = "finish"
    }

    class NavigateCommand {
        # commandWord = "nav"
    }

    class ShowQuotesCommand {
        # commandWord = "show"
    }
    
    class DeleteQuoteCommand {
        # commandWord = "delete"
    }

    abstract class Command {
        # commandWord
        + {abstract} void execute()
        + boolean isExit(): false
    }
}

package ui {

}

package data {

}


Command <|-- AddItemCommand 
Command <|-- AddQuoteCommand
Command <|-- ExitCommand
Command <|-- DeleteItemCommand
Command <|-- RegisterCommand
CalculateTotalCommand --|> Command
FinishQuoteCommand --|> Command
NavigateCommand --|> Command
DeleteQuoteCommand --|> Command
ShowQuotesCommand --|> Command

command .> ui
command ..> data

@enduml
